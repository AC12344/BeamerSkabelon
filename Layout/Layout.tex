\newcommand{\executeiffilenewer}[3]{%
\ifnum\pdfstrcmp{\pdffilemoddate{#1}}%
{\pdffilemoddate{#2}}>0%
{\immediate\write18{#3}}\fi%
}
\newcommand{\includesvg}[2][./]{%
\executeiffilenewer{#1#2.svg}{#1#2.pdf}
{inkscape -z -D --file=#1#2.svg --export-pdf=#1#2.pdf --export-latex}
\import{#1}{#2.pdf_tex}
}
% ex. \includesvg[images/]{image} Trailing slash er vigtigt for stien
% Kilde: http://ctan.math.utah.edu/ctan/tex-archive/info/svg-inkscape/InkscapePDFLaTeX.pdf
% Størrelse kan blive sat med \def\svgwidth{<size>} eller \def\svgscale{<scale>} umiddelbart
% før \includesvg

\usepackage{tikz}
\definecolor{color1}{RGB}{193,193,204}
\definecolor{color2}{RGB}{255,255,255}

\newcounter{totavalue}

\def\aux{1}
\def\radius{19pt}
\def\thickness{2pt}

\newcommand\circcounteri{%
\ifnum\inserttotalframenumber<2\relax
\else
  \setcounter{totavalue}{\inserttotalframenumber}%
\fi%
\pgfmathsetmacro{\aux}{360/\thetotavalue}%
\begin{tikzpicture}[overlay]
	\fill[color1]
		circle (\radius+1pt);
	\fill[color2]
		circle (\radius-1pt-\thickness);
	\fill[black] 
		(0:\radius)
		arc  (0:\insertframenumber*\aux:\radius)
		arc  (\insertframenumber*\aux:180+\insertframenumber*\aux:\thickness/2)
		arc  (\insertframenumber*\aux:0:\radius-\thickness) -- cycle;
	\node[inner sep=0pt] at (0,0) 
		{\includegraphics[width=\logoscale]{Layout/SDU_BLACK.pdf}};
	\fill[color1]
		(0:\radius) circle(5pt);
	\fill[color2]
		(0:\radius) circle(4pt);
	\node[font=\color{color3}] at (\radius,0) {\insertframenumber};
\end{tikzpicture}%
}
\addtobeamertemplate{headline}{}{\vskip22pt\hfill\circcounteri\hspace*{25pt}}
